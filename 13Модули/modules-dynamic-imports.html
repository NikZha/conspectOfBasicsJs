Динамические импорты

Запрещено импортировать модули из функции: 

    
import ... from getModuleName(); // Ошибка, должна быть строка
Запрещено вставлять import в блок скобок { } или if: 
    
if(...) {
  import ...; // Ошибка, запрещено
}

{
  import ...; // Ошибка, мы не можем ставить импорт в блок
}

Выражение import()

Суть выражения импорт - получить string и загрузить его: 

    
let modulePath = prompt("Какой модуль загружать?");

import(modulePath)
  .then(obj => (объект модуля))
  .catch(err => (ошибка загрузки, например если нет такого модуля))
Как видно это ассинхронная функциия, так же: 
    
// 📁 say.js
export function hi() {
  alert(`Привет`);
}

export function bye() {
  alert(`Пока`);
}
Импорт будет таким: 
    
let {hi, bye} = await import('./say.js');

hi();
bye();
если указан export default, то: 
    
// 📁 say.js
export default function() {
  alert("Module loaded (export default)!");
}
//Импорт в другом файле:
let obj = await import('./say.js');
let say = obj.default;
// или, одной строкой: let {default: say} = await import('./say.js');

say();
Так же динамический импорт можно использовать в ф-ции: 

    
        //в блоке скрипт следующая ф-ция:
    async function load(params) {
        let say = await import('./say.js')
        say.sayHi('Test')
        say.sayBye('Test')
        say.default()
    }
Как видно все работает без указания в script свойства module. По мимо такой ф-ции, можно вызвать в блоках .then и .catch: 

    
    async function loadThen(params) {
        await import('./say.js').then(obj => {
            obj.sayHi('Then');
            obj.sayBye('Then');
            obj.default()
        }).catch(err => alert(err))
    }
