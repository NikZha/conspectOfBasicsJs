    Добавление и удаление элементов

    • arr.push(...items) – добавляет элементы в конец,
    • arr.pop() – извлекает элемент из конца,
    • arr.shift() – извлекает элемент из начала,
    • arr.unshift(...items) – добавляет элементы в начало.

    splice

    Если удалять при помощи delete остаются дырки в массиве: 

        
let arr = ["I", "go", "home"];
delete arr[1]; // удалить "go"
alert( arr[1] ); // undefined
// теперь arr = ["I",  , "home"];
alert( arr.length ); // 3
    Выход - заменить splice: 
        
arr.splice(start[, deleteCount, elem1, ..., elemN])

let arr = ["Я", "изучаю", "JavaScript"];
arr.splice(1, 1); // начиная с индекса 1, удалить 1 элемент
alert( arr ); // осталось ["Я", "JavaScript"]

let arr = ["Я", "изучаю", "JavaScript", "прямо", "сейчас"];
// удалить 3 первых элемента и заменить их другими
arr.splice(0, 3, "Давай", "танцевать");
alert( arr ) // теперь ["Давай", "танцевать", "прямо", "сейчас"]
    Эта ф-ция возвращает удалённый массив: 
        
let arr = ["Я", "изучаю", "JavaScript", "прямо", "сейчас"];
// удалить 2 первых элемента
let removed = arr.splice(0, 2);
alert( removed ); // "Я", "изучаю" <-- массив из удалённых элементов
    Так же можно не удалять оставить второй агрумент в 0: 
        
let arr = ["Я", "изучаю", "JavaScript"];
// с индекса 2
// удалить 0 элементов
// вставить "сложный", "язык"
arr.splice(2, 0, "сложный", "язык");
alert( arr ); // "Я", "изучаю", "сложный", "язык", "JavaScript"
    В этом и в других методах массива допускается использование отрицательных индексов. Они определяют позицию с конца массива, как тут: 
        
let arr = [1, 2, 5];
// начиная с индекса -1 (перед последним элементом)
// удалить 0 элементов,
// затем вставить числа 3 и 4
arr.splice(-1, 0, 3, 4);
alert( arr ); // 1,2,3,4,5

    slice

    Ф-ция slice просто возвращает массив между двумя аргументами: 

                
let arr = ["t", "e", "s", "t"];

alert( arr.slice(1, 3) ); // e,s (копирует с 1 до 3)

alert( arr.slice(-2) ); // s,t (копирует с -2 до конца)

    concat

    Ф-ция сливает массивы: 

        
let arr = [1, 2];

// создать массив из: arr и [3,4]
alert( arr.concat([3, 4]) ); // 1,2,3,4

// создать массив из: arr и [3,4] и [5,6]
alert( arr.concat([3, 4], [5, 6]) ); // 1,2,3,4,5,6

// создать массив из: arr и [3,4], потом добавить значения 5 и 6
alert( arr.concat([3, 4], 5, 6) ); // 1,2,3,4,5,6
    Так же добавляет объекты к массивам:
    Но если массивоподобный объект имеет специальное свойство Symbol.isConcatSpreadable, то он обрабатывается как массив, с помощью concat: вместо него добавляются его элементы: 
        
let arr = [1, 2];

let arrayLike = {
  0: "что-то",
  1: "ещё",
  [Symbol.isConcatSpreadable]: true,
  length: 2
};

alert( arr.concat(arrayLike) ); // 1,2,что-то,ещё

    Поиск в массиве

    Теперь рассмотрим методы поиска в массиве.

    indexOf/lastIndexOf и includes

    • arr.indexOf(item, from) ищет item начиная с индекса from и возвращает номер индекса, на котором был найден искомый элемент, в противном случае -1.
    • arr.includes(item, from) ищет item начиная с индекса from и возвращает true, если поиск успешен.
    Обычно не указывают позицию поиска, например: 
                
let arr = [1, 0, false];
alert( arr.indexOf(0) ); // 1
alert( arr.indexOf(false) ); // 2
alert( arr.indexOf(null) ); // -1
alert( arr.includes(1) ); // true
    Методы используют строгое сравнние === а следовательно ищеи 0, а не false и наоборот.
    Метод arr.lastIndexOf похож на indexOf, но ищет справа налево. То бишь с конца.

    Метод includes правильно обрабатывает NaN

                
const arr = [NaN];
alert( arr.indexOf(NaN) ); // -1 (неверно, должен быть 0)
alert( arr.includes(NaN) );// true (верно)

    find и findIndex/findLastIndex

    Синтаксис: 

                
let result = arr.find(function(item, index, array) {
  // если true - возвращается текущий элемент и перебор прерывается
  // если все итерации оказались ложными, возвращается undefined
});
    В реале используется с одним аргументом: 
                
let users = [
  {id: 1, name: "Вася"},
  {id: 2, name: "Петя"},
  {id: 3, name: "Маша"}
];

let user = users.find(item => item.id == 1);
    У метода arr.findIndex такой же синтаксис, но он возвращает индекс, на котором был найден элемент, а не сам элемент. Значение -1 возвращается, если ничего не найдено.
    Метод arr.findLastIndex похож на findIndex, но ищет справа налево, наподобие lastIndexOf.

    filter

    Возвращает массив отфильтрованный по каким-то критериям. Синтаксис: 

                        
let results = arr.filter(function(item, index, array) {
  // если `true` -- элемент добавляется к results и перебор продолжается
  // возвращается пустой массив в случае, если ничего не найдено
});
    Работает примерно так же: 
        
let users = [
  {id: 1, name: "Вася"},
  {id: 2, name: "Петя"},
  {id: 3, name: "Маша"}
];

// возвращает массив, состоящий из двух первых пользователей
let someUsers = users.filter(item => item.id < 3);

    Преобразование массива

    Методы преобразования массива:

    map

    map один из методов преобразования множества о котором ты читал в Грокаем алгоритмы. Преобразует каждый элемент множетства, синтаксис: 

                        
let result = arr.map(function(item, index, array) {
  // возвращается новое значение вместо элемента
});
//Например преобразование массива стрингов в массив интов на основе длинны стрингов:
let lengths = ["Бильбо", "Гэндальф", "Назгул"].map(item => item.length);

    sort

    sort сортировка массива на основе быстрой сортировки. Сортирует массив в лексикографическом порядке, то есть: 

                    
let arr = [ 1, 2, 15 ];
// метод сортирует содержимое arr
arr.sort(); // 1, 15, 2
    Для нормальной работы надо сделать ф-цию на основе которой идёт сравнение, например: 
                
function compareNumeric(a, b) {
  if (a > b) return 1;
  if (a == b) return 0;
  if (a < b) return -1;
}

let arr = [ 1, 2, 15 ];

arr.sort(compareNumeric); 
alert(arr);  // 1, 2, 15
    Но можно ещё упросить ф-цию: 
        
let arr = [ 1, 2, 15 ];

arr.sort(function(a, b) { return a - b; });
    Автор учебника предлагает стрелочную функцию: 
        
arr.sort( (a, b) => a - b );
    Судя по дипсику - стрелочная ф-ция это буквально официальное название.

    Второй момент - стрелочная функция не всегда замыкание - эта например не замыкание потому что не использует внешнего контекста, то есть не запоминает никаких переменных. Например замыканиями были бы: 
        
const arr = [1, 2, 3];        
let threshold = 0;

arr.sort((a, b) => {
  threshold++; // Изменяет внешнюю переменную → замыкание!
  return a - b - threshold;
});        
//Или такой случай:
const compareWithThreshold = (threshold) => {
  return (a, b) => a - b - threshold; // Использует внешний `threshold` → замыкание!
};
arr.sort(compareWithThreshold(10));
    То есть в приведённых выше примерах есть использование внешних данных, а arr.sort( (a, b) => a - b ); просто колбек.
    Для сортировки строк лучше использовать localeCompare, синтаксис такой: 
        
let countries = ['Österreich', 'Andorra', 'Vietnam'];
alert( countries.sort( (a, b) => a.localeCompare(b) ) ); // Andorra,Österreich,Vietnam (правильно!)

let arr = [1, 2, 3, 4, 5];
arr.reverse(); // 5,4,3,2,1

    split and join

    Весьма интересные ф-ции которые позволяют преобразовать sting to array и наоборот, рассматрим примеры: 

                        
let names = 'Вася, Петя, Маша';

let arr = names.split(', '); //разделение имён по вставленным в кавычки символам. В итоге получился массив
    У метода split есть необязательный второй числовой аргумент – ограничение на количество элементов в массиве. Если их больше, чем указано, то остаток массива будет отброшен. На практике это редко используется: 
                    
let arr = 'Вася, Петя, Маша, Саша'.split(', ', 2); // Вася, Петя
    Если будет ничто в кавычках, то произойдёт создание массива посимвольно: 
        
let str = "тест"; // т,е,с,т

    Метод join создаёт из массива строку, например: 

                    
let arr = ['Вася', 'Петя', 'Маша'];

    reduce/reduceRight

    Как уже упоминалось ранее - классический метод, который упоминался в книге грокаем алгоритмы. Вычисляет одно значение на основе массива: 

                        
let value = arr.reduce(function(accumulator, item, index, array) {
  // ...
}, [initial]);
    Аргументы:
    • accumulator – результат предыдущего вызова этой функции, равен initial при первом вызове (если передан initial),
    • item – очередной элемент массива,
    • index – его позиция,
    • array – сам массив.
    В целом не очевидная ф-ция: 
        
let arr = [1, 2, 3, 4, 5];
let result = arr.reduce((sum, current) => sum + current, 0);
alert(result); // 15
    Мы также можем опустить начальное значение: 
        
let arr = [1, 2, 3, 4, 5];
// убрано начальное значение (нет 0 в конце)
let result = arr.reduce((sum, current) => sum + current);
alert( result ); // 15
    Но такое использование требует крайней осторожности. Если массив пуст, то вызов reduce без начального значения выдаст ошибку. 
        
let arr = [];
// Error: Reduce of empty array with no initial value
// если бы существовало начальное значение, reduce вернул бы его для пустого массива.
arr.reduce((sum, current) => sum + current);
    На самом деле такой код упал в ошибку с сообщением:
    Uncaught TypeError TypeError: Reduce of empty array with no initial value (и далее)

    Array.isArray

    Сама ф-ция работает как Array.isArray([]) в таком виде: 

                
let arr = [2,2,3]
let obj = {
    2:2,
    3:3
}
console.log(`arr is ${typeof arr} and obj is ${typeof obj}`) //arr is object and obj is object
console.log(Array.isArray(arr)) //true
alert(Array.isArray({})); // false
    Как понятно из кода typeof не делает различие между объектом и массивом (потому что это одно и тоже в JavaScript), а потому нужен такой хитрый огород чтобы их отличать Ссылка на код.

    Большинство методов поддерживают «thisArg»

    Почти все методы массива, которые вызывают функции – такие как find, filter, map, за исключением метода sort, принимают необязательный параметр thisArg. 

                    
arr.find(func, thisArg);
arr.filter(func, thisArg);
arr.map(func, thisArg);
// ...
// thisArg -- необязательный последний аргумент
    Значение параметра thisArg становится this для func.
    Например, тут мы используем метод объекта army как фильтр, и thisArg передаёт ему контекст: 
        
let army = {
  minAge: 18,
  maxAge: 27,
  canJoin(user) {
    return user.age >= this.minAge && user.age < this.maxAge;
  }
};

let users = [
  {age: 16},
  {age: 20},
  {age: 23},
  {age: 30}
];

// найти пользователей, для которых army.canJoin возвращает true
let soldiers = users.filter(army.canJoin, army);
    Если бы мы в примере выше использовали просто users.filter(army.canJoin), то вызов army.canJoin был бы в режиме отдельной функции, с this=undefined. Это тут же привело бы к ошибке.
    Вызов users.filter(army.canJoin, army) можно заменить на users.filter(user => army.canJoin(user)), который делает то же самое. Последняя запись используется даже чаще, так как функция-стрелка более наглядна.

function Calculator() {
  this.operations = {
    '+': (a, b) => a + b,
    '-': (a, b) => a - b
  };

  this.addMethod = function(operator, func) {
    this.operations[operator] = func;

  };
}
