Задача решена так:


        let ask = (
            question = 'Вы согласны?',
            yes = () => {
                alert("Вы согласились.");
            },
            no = () => {
                alert("Вы отменили выполнение.");
            }
        ) => {
            if (confirm(question)) yes()
            else no();
        }
