Декораторы и переадресация вызова, call/apply

В этом разделе будет рассмотрена возможность создания кеширующих объектов, коротые захватывают аргументы других ф-ций и хранят их работу.

Прозрачное кеширование

Предположим, что функция ведёт себя, как математическая ф-ция получает значения из одного множества и возвращает значения из другого множества, получая один и теже значения, возвращает такие же результаты. И если вычисления связаны с огромной работой, то возникает потребность кеширования результатов: 

                
function fact(x) {
    return x == 1 ? 1 : x*fact(x-1) 
}

function cachingDecorator(func) {
  let cache = new Map();

  return function(x) {
    if (cache.has(x)) {    // если кеш содержит такой x,
      return cache.get(x); // читаем из него результат
    }

    let result = func(x); // иначе, вызываем функцию

    cache.set(x, result); // и кешируем (запоминаем) результат
    return result;
  };
}
fact = cachingDecorator(fact);
console.log(fact(20))
console.log(fact(20))
Функция выбрана глупая с рекурсией и результаты её работы закешированы и это работает.
Таким образом:
  • Функцию cachingDecorator можно использовать повторно. Мы можем применить её к другой функции.
  • Логика кеширования является отдельной, она не увеличивает сложность самой slow (если таковая была).
  • При необходимости мы можем объединить несколько декораторов (речь об этом пойдёт позже).

Применение «func.call» для передачи контекста

Приведённый выше декоратор не работает с методами объектов, например: 

    
let worker = {
    beginValue(){
        return 1;
    },
    slow(x) {
        console.log(`Recursion begin:${x}`)
    return x == 1 ? 1 : x*this.slow(x-1) 
}
};

function cachingDecorator(func) {
  let cache = new Map();
  return function(x) {
    if (cache.has(x)) {
      return cache.get(x);
    }
    let result = func(x); // (**)
    cache.set(x, result);
    return result;
  };
}

console.log(worker.slow(5))
worker.slow = cachingDecorator(worker.slow); // теперь сделаем его кеширующим
console.log(worker.slow(5)) //ошибка
По причине что у декоратора получается this = undefined.

Существует специальный вызов: 

        
func.call(context, arg1, arg2, ...)
Он запускает функцию func, используя первый аргумент как её контекст this, а последующие – как её аргументы.

Надо заметить, что вызов .call это вызов к любой функции, таким образом можно исправить исходный пример: 

            
let worker = {
    beginValue(){
        return 1;
    },
    slow(x) {
        console.log(`Recursion begin:${x}`)
    return x == 1 ? 1 : x*this.slow(x-1) 
}
};

function cachingDecorator(func) {
  let cache = new Map();
  return function(x) {
    if (cache.has(x)) {
      return cache.get(x);
    }
    let result = func.call(this, x); // func здесь имя переданной функции, а .call просто вызов передающий this и агрумент
    cache.set(x, result);
    return result;
  };
}

console.log(worker.slow(5))
worker.slow = cachingDecorator(worker.slow); // теперь сделаем его кеширующим
console.log(worker.slow(5)) //ошибка

Переходим к нескольким аргументам с «func.apply»

Можно сделать декораторы более универсальными, и работающими с несколькимим ф-циями: 

            
let worker = {
  slow(min, max){
    return max > min ? max : min;
  }
}

function cachingDecorator(func, hash) {
  let cache = new Map();
  return function() {
    let key = hash(arguments); // (*)
    if (cache.has(key)) {
      return cache.get(key);
    }

    let result = func.call(this, ...arguments); // (**) Принимает ссылку на this и создаёт массив из агрументов.

    cache.set(key, result);
    return result;
  };
}

function hash(args) { //* функия очень простая возвращает string из двух аргументов, как ты понимаешь можно больше.
  return args[0] + ',' + args[1];
}

worker.slow = cachingDecorator(worker.slow, hash);

console.log( worker.slow(3, 5) ); // работает
console.log( "Again " + worker.slow(3, 5) ); // аналогично (из кеша)

Накидал такое в качестве функции для ключей мапы. Почитав ниже, я вспомнил метод .join() c которым этот огород нахер не нужен. 
        
function hash1(args){
  let str = '' + args[0] + ',';
  for(let i=1;i<args.length;i++) str+=args[i]+',';
  
}

Для принятия массива аргументов есть другая встроенная функция: 

    
func.apply(context, args)
С одним отличием: А apply принимает только псевдомассив args.

Заимствование метода

Но когда массив является псевмассивом, как arguments, то возникнет ошибка: 

    
        function hash() {
            return arguments.join()
        }

Тем не менее, есть простой способ использовать соединение массива: 

    
function hash1() {
  alert( [].join.call(arguments) ); // 1,2
}

Этот трюк называется заимствование метода.

Мы берём (заимствуем) метод join из обычного массива [].join. И используем [].join.call, чтобы выполнить его в контексте arguments.

Итого

Декоратор – это обёртка вокруг функции, которая изменяет поведение последней. Основная работа по-прежнему выполняется функцией.

Работу по прозначному декорированию позволяют выполнить методы .call and .aply: первый работает с аргументами, второй с псевдомассивом.

Так же был рассмотрен метод заимствования методов когда мы вызываем метод у объекта в контексте другого объекта. Весьма распространено заимствовать методы массива и применять их к arguments. В качестве альтернативы можно использовать объект с остаточными параметрами ...args, который является реальным массивом.
